Traffic Management

Optimize traffic signal timings at a four-way intersection to balance arrival and departure rates, minimizing congestion. Ensure green times respect minimum and maximum constraints for smooth traffic flow.

Objectives for optimize green signal timings

  • Minimize queue lengths at each approach to reduce traffic congestion.
  • Respect operational constraints, such as: Minimum green time to ensure safety and traffic flow and Maximum green time to prevent excessive delays for other approaches.
  • Achieve stability in signal timings through iterative adjustments.

Implementation

Our implementation utilizes advanced data structures such as Arrays, Linked Lists, Hash Tables, Priority Queues and Vectors to optimize green signal timings defficiently. These tools ensure real-time data access and efficient processing tailored to the Traffic Management.

using namespace std;

struct Approach {
    double arrival_rate;
    double departure_rate;
    int green_time;
    int queue_length;
};

void initialize_green_times(vector& approaches, int cycle_time, int min_green, int max_green) {
    double total_demand = 0.0;
    for (const auto& approach : approaches) {
        total_demand += approach.arrival_rate;
    }
    for (auto& approach : approaches) {
        approach.green_time = static_cast((approach.arrival_rate / total_demand) * cycle_time);
        approach.green_time = max(approach.green_time, min_green);
        approach.green_time = min(approach.green_time, max_green);
    }
}

void calculate_queue_lengths(vector& approaches) {
    for (auto& approach : approaches) {
        approach.queue_length = max(0, static_cast(approach.arrival_rate * approach.green_time - approach.departure_rate * approach.green_time));
    }
}

void adjust_green_times(vector& approaches, int min_green, int max_green) {
    priority_queue> pq; // Max-heap to prioritize approaches with the largest queues

    for (int i = 0; i < approaches.size(); ++i) {
        pq.push({approaches[i].queue_length, i});
    }

    while (!pq.empty()) {
        auto [queue_length, idx] = pq.top();
        pq.pop();

        if (queue_length > 0 && approaches[idx].green_time < max_green) {
            approaches[idx].green_time++;
        } else if (queue_length < 0 && approaches[idx].green_time > min_green) {
            approaches[idx].green_time--;
        }

        approaches[idx].green_time = max(approaches[idx].green_time, min_green);
        approaches[idx].green_time = min(approaches[idx].green_time, max_green);
    }
}

bool check_convergence(const vector& approaches, const vector& prev_green_times) {
    for (int i = 0; i < approaches.size(); ++i) {
        if (approaches[i].green_time != prev_green_times[i]) {
            return false;
        }
    }
    return true;
}

void optimize_traffic_signals(vector& approaches, int cycle_time, int min_green, int max_green) {
    initialize_green_times(approaches, cycle_time, min_green, max_green);

    bool converged = false;
    while (!converged) {
        vector prev_green_times;
        for (const auto& approach : approaches) {
            prev_green_times.push_back(approach.green_time);
        }

        calculate_queue_lengths(approaches);
        adjust_green_times(approaches, min_green, max_green);

        converged = check_convergence(approaches, prev_green_times);
    }
}

int main() {
    vector approaches = {
        {0.2, 0.25, 0, 0},
        {0.15, 0.2, 0, 0},
        {0.1, 0.15, 0, 0},
        {0.05, 0.1, 0, 0}
    };

    int cycle_time = 60;
    int min_green = 10;
    int max_green = 30;

    optimize_traffic_signals(approaches, cycle_time, min_green, max_green);

    cout << "Optimized green times:" << endl;
    for (int i = 0; i < approaches.size(); ++i) {
        cout << "Approach " << i + 1 << ": " << approaches[i].green_time << " seconds" << endl;
    }

    return 0;
}
